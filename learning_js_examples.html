Exercise 4: Functions

Write calculateAge() and describePerson() to encapsulate the logic.

// exercise4.js
function calculateAge(year) {
  return 2025 - year;
}

function describePerson(firstName, lastName, birthYear) {
  let age = calculateAge(birthYear);
  console.log(`Hello, my name is ${firstName} ${lastName} and I am ${age} years old.`);

  if (age < 13) {
    console.log("You are a child.");
  } else if (age >= 13 && age <= 19) {
    console.log("You are a teenager.");
  } else {
    console.log("You are an adult.");
  }
}

describePerson("Parham", "Firoozi", 2000);

Exercise 8: Methods in Objects

Add a getSummary() method inside the object.

let favoriteMovie = {
  title: "Intersex",
  director: "Bokon Nanat",
  year: 2000,
  genres: ["kir", "kos", "kun"],

  getSummary: function() {
    return `${this.title} (released in ${this.year}) is a ${this.genres[0]}/${this.genres[1]} film directed by ${this.director}.`;
  }
};

console.log(favoriteMovie.getSummary());

Project 1: Random Background Color

Click a button to change the page’s background to a random color.





  
Click to Change Background Color

  
  




function changeBackground() {
  const colors = ["#FF5733", "#33FFBD", "#3380FF", "#F333FF", "#A1FF33"];
  const randomIndex = Math.floor(Math.random() * colors.length);
  document.body.style.backgroundColor = colors[randomIndex];
}

Project 2: Dark Mode Toggle

Switch between light and dark themes by toggling a CSS class on <body>.





  


  
Dark Mode Toggle

  
  




function toggleMode() {
  const body = document.body;
  if (body.classList.contains("dark")) {
    body.classList.replace("dark", "light");
  } else {
    body.classList.replace("light", "dark");
  }
}

Project 3: Mini To-Do List

Add tasks via an input, and append them to a list. Click to remove.





  
My To-Do List

  
  
  
    
  




function addTask() {
  let input = document.getElementById("taskInput");
  let text = input.value.trim();
  if (text !== "") {
    let li = document.createElement("li");
    li.textContent = text;
    li.onclick = () => li.remove(); // click to remove
    document.getElementById("taskList").appendChild(li);
    input.value = "";
  }
}
