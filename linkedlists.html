class Car:
    def __init__(self, brand, model):
        self.brand = brand
        self.model = model

    def drive(self):
        print(f"{self.brand} {self.model} is driving.")

    def stop(self):
        print(f"{self.brand} {self.model} has stopped.")

# Usage
my_car = Car("Toyota", "Corolla")
my_car.drive()
my_car.stop()

friend_car = Car("Honda", "Civic")
friend_car.drive()
